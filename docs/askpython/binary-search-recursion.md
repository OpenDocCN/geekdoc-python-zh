        print("ERROR!")

    # 2\. If element is not found lb exceeds ub    
    elif(lb>ub):
        print("Not found!")

    # 3\. Keep searching for the element in array

        mid = int((lb+ub)/2)
        if(arr[mid]==X):
            print(mid+1)
        elif(arr[mid]>X):
            Binary_Search(arr,n,lb,mid,X);
        elif(arr[mid]<X):
            Binary_Search(arr,n,mid+1,ub,X);
